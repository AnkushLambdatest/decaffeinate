var { pow, sqrt } = Math;
var sq = (x) => pow(x, 2);

// Represents a 2D point.
class Point {
  constructor(x, y) {
    this.x = x, this.y = y;
  }

  distanceTo({x, y}) {
    var xd = sq(x - this.x);
    var yd = sq(y - this.y);
    return sqrt(xd + yd);
  }
}
